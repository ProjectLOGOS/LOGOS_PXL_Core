# docker-compose.yml and PYTHONPATH includes the correct paths:
#
# volumes:
#   - ./external_libraries:/app/external_libraries:ro
# environment:
#   - PYTHONPATH=/app:/app/external_libraries/library-name
#
# This approach provides:
# 1. Faster container builds (no pip installs of large packages)
# 2. Consistent library versions across all services
# 3. Easy library updates by replacing external_libraries directory
# 4. Reduced container image sizes
